document.getElementById("checkout-form").addEventListener("submit", function (e) { e.preventDefault(); const form = e.target; const name = form.fullname.value.trim(); const email = form.email.value.trim(); const address = form.address.value.trim(); const city = form.city.value.trim(); const zip = form.zip.value.trim(); if (!name || !email || !address || !city || !zip) { alert("Please fill in all fields."); return; } if (!/^\d{6}$/.test(zip)) { alert("Zip code must be 6 digits."); return; } alert("Order placed successfully! Thank you."); form.reset(); localStorage.removeItem("cart"); window.location.href = "confirmation.html"; });
